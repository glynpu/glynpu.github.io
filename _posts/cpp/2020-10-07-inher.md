无论是内部定义属性/还是继承
class 默认是private
struct 默认是public



```
#include <iostream>

using namespace std;
class Person {
 public:
  int age = 10;
};

class Student : Person {
 public:
  void info() { cout << "student's age: " << age << endl; }
};

struct Teacher : Person {
 // 写不写public 都一样，struct 默认就是public
 // public:
  void info() { cout << "teacher's age: " << age << endl; }
};

int main() {
  Student s;
  s.info();
  // cannot cast Student to its private base class Person
  // cout << s.age << endl;
  Teacher t;
  t.info();
  cout << t.age << endl;
}
```
