1. sort 函数需要custom class 支持operator <
2. 能不用frined 函数就不要用。frined 函数可以访问私有变量，破坏了封装性

c++的两个启示：
1. 朋友并不是越多越好(友元函数破坏封装性)
2. 朋友比孩子关系更密切(友元函数可以访问私有变量，但是继承类不可以) 




```
#include <algorithm>
#include <iostream>
#include <vector>

using namespace std;

class Person {
 public:
  // 这个无参构造函数是为了给子类构造时用的
  // 如果基类没有无参构造函数，子类要显示调用
  // Student() : Person(10){};
  // Person(){};
	 Person() = default;



  // explicit 禁止隐式类型转换
  explicit Person(int distance) : distance(distance) {}
  bool operator<(const Person& p) { return abs(distance) < abs(p.distance); }
  // virtual int info() const { return distance; }
  virtual int Info() const { return distance; }
  // 友元函数可以访问类的私有成员
  /* friend ostream& operator<<(ostream& os, const Person& p); */

 private:
  int distance;
};

/* ostream& operator<<(ostream& os, const Person& p) { */
/*   /1* p.distance += 1000; *1/ */
/*   os << p.distance; */
/*   return os; */
/* } */
ostream& operator<<(ostream& os, const Person& p) {
  // 输入是const Person, 下面一行会报： distance is a private member of Person
  // 友元函数可能保证该函数能够访问成员变量
  // const 使其只能读，不能写
  /* p.distance += 1000; */
  os << p.info();
  return os;
}
 auto a = info()

class Student : public Person {
 public:
  Student(){};
  virtual int info() const {
    // distance is private member of 'Person'
    /* cout << distance << endl; */
    return 1234;
  }
};

template <class T>
void show_vector(std::vector<T> tv) {
  for (auto t : tv) {
    cout << t << " ";
  }
  cout << endl;
}
// 模版函数instantiate
template void show_vector(std::vector<Person> tv);

int main() {
  int max = 10;
  int i = 0;
  int secret;
  std::vector<Person> pv;

  while (i++ < max) {
    secret = rand() % 100 - 50;
    pv.push_back(Person(secret));
  }
  show_vector(pv);

  sort(pv.begin(), pv.end());
  show_vector(pv);

  Student s;
  cout << s.info() << endl;

  return 0;
}

```
